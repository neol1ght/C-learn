#include <iostream>
#include <string>
using namespace std;

// 5.1-5.2: Базовый класс Book
class Book {

public:
    string title;
    string author;
    int year;
    // Конструктор
    Book(string t = "", string a = "", int y = 0) : title(t), author(a), year(y) {}

    // Метод для вывода информации о книге
    virtual void showInfo() {
        cout << "Название: " << title << endl;
        cout << "Автор: " << author << endl;
        cout << "Год издания: " << year << endl;
    }
};

// 5.3-5.4: Производный класс BookWithGenre
class BookWithGenre : public Book {

public:
    string genre;  // 5.3: новое поле - жанр
    // 5.4: Переопределенный конструктор
    BookWithGenre(string t = "", string a = "", int y = 0, string g = "")
        : Book(t, a, y), genre(g) {}

    // 5.4: Переопределенный метод
    void showInfo() override {
        cout << "Название: " << title << endl;
        cout << "Автор: " << author << endl;
        cout << "Год издания: " << year << endl;
        cout << "Жанр: " << genre << endl;
    }
};

// 5.5: Главная функция программы
int main() {
    setlocale(LC_ALL, "Russian");
    string title, author, genre;
    int year;

    // Запрос данных у пользователя
    cout << "Введите данные о книге:" << endl;
    cout << "Название: ";
    getline(cin, title);

    cout << "Автор: ";
    getline(cin, author);

    cout << "Год издания: ";
    cin >> year;

    cout << "Жанр: ";
    getline(cin, genre);

    // Создание объекта-потомка
    BookWithGenre myBook(title, author, year, genre);

    // Вывод полных данных о книге
    cout << "\nПолная информация о книге:" << endl;
    cout << "---------------------------" << endl;
    myBook.showInfo();

    return 0;
}
