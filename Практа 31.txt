
#include <iostream>
#include <string>
using namespace std;

// Определяющий класс
class GraphicObject {

public:
    string name;
    GraphicObject(string objName) : name(objName) {}

    virtual double area() {
        return 0;
    }

    virtual void show() {
        cout << name << " площадь: " << area() << endl;
    }
};

// Класс Круг - наследовался от GraphicObject
class Circle : public GraphicObject {

public:
    double r;
    Circle(double radius) : GraphicObject("Круг"), r(radius) {}

    double area() override {
        return 3.14 * r * r;
    }
};

// Класс Квадрат - наследовался от GraphicObject
class Square : public GraphicObject {
private:
    double a;

public:
    Square(double side) : GraphicObject("Квадрат"), a(side) {}

    double area() override {
        return a * a;
    }
};

int main() {
    setlocale(LC_ALL, "Russian");
    double radius, side;

    cout << "Введите радиус круга: ";
    cin >> radius;

    cout << "Введите сторону квадрата: ";
    cin >> side;

    Circle circle(radius);
    Square square(side);

    GraphicObject* obj1 = &circle;
    GraphicObject* obj2 = &square;

    obj1->show();
    obj2->show();

    return 0;
}
