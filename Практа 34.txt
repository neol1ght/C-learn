#include <iostream>
#include <list>
#include <algorithm>
using namespace std;

// Основная работа функции
int main() {
    setlocale(LC_ALL, "Russian");
    list<int> numbers;
    int n, value;

    // Запрос элементов
    cout << "Введите количество элементов: ";
    cin >> n;

    // Вводим элементы
    cout << "Введите " << n << " целых чисел:\n";
    for (int i = 0; i < n; ++i) {
        cin >> value;
        // В контейнер-список
        numbers.push_back(value);
    }

    // Подсчет четных элементов с помощью алгоритма (библиотека algorithm) count_if
    int evenCount = count_if(numbers.begin(), numbers.end(),
        [](int x) { return x % 2 == 0; });

    // Вывод результата
    cout << "Количество четных элементов: " << evenCount << endl;

    return 0;
}
